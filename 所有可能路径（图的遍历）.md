### 所有可能路径
```
class Solution:
    def __init__(self):
        self.res = []
        
    def allPathsSourceTarget(self, graph: List[List[int]]) -> List[List[int]]:
        path = []
        self.traverse(graph, 0, path)
        return self.res

    def traverse(self, graph, s, path):
        ## 添加节点s到路径
        path.append(s)
        n = len(graph)
        if s == (n-1):
            ## 到达终点
            self.res.append(path+[])
            path.pop()
            return

        ## 递归每个相邻节点
        for v in graph[s]:
            self.traverse(graph, v, path)

        ## 从路径移出节点s
        path.pop()
        
```
