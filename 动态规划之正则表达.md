### 正则表达式匹配
```

class Solution:
    def isMatch(self, s: str, p: str) -> bool:
        # 计算 p[j..] 是否匹配 s[i..]
        memo = {}
        def dp(s, i, p, j):
            m = len(s)
            n = len(p)
            # base case
            if j == n:
                return i == m
            if i == m:
                if (n-j)%2 == 1:
                    return False
                while j+1<n:
                    if p[j+1] != '*':
                        return False
                    j += 2
                return True

            # 记录状态 (i, j)，消除重叠子问题
            key = str(i) + "," + str(j)
            if memo.__contains__(key):
                return memo[key]

            res = False

            if s[i] == p[j] or p[j] == '.':
                if j< n - 1 and p[j+1] == '*':
                    res = dp(s,i,p,j+2) or dp(s,i+1,p,j)
                else:
                    res = dp(s,i+1,p,j+1)
            else:
                if j<n-1 and p[j+1] == '*':
                    res = dp(s,i,p,j+2)
                else:
                    res = False
            # 将当前结果记入备忘录
            memo[key] = res
            return res
        return dp(s,0,p,0)
```        

```
class Solution:
    def isMatch(self, s: str, p: str) -> bool:
        self.memo = {}
        return self.dp(s,0,p,0)

    def dp(self, s, i, p, j):
        m = len(s)
        n = len(p)
        # base case
        if j == n:
            return i == m
        if i == m:
            if (n-j)%2 == 1:
                return False
            else:
                while j+1<n:
                    if p[j+1] != '*':
                        return False
                    else:
                        j += 2
                return True

        # 记录状态（i,j） 消除重叠子问题
        key = str(i) + ',' + str(j)
        if key in self.memo:
            return self.memo[key]

        res = False
        
        # 匹配
        if s[i] == p[j] or p[j] == '.':
            if j < n - 1 and p[j+1] == '*':
                # 1.1 通配符匹配0次或者多次
                res = self.dp(s,i,p,j+2) or self.dp(s,i+1,p,j)
            else:
                # 1.2 常规匹配1次
                res = self.dp(s,i+1,p,j+1)
        # 不匹配
        else:
            if j < n - 1 and p[j+1] == '*':
                # 2.1 通配符匹配0次
                res = self.dp(s,i,p,j+2)
            else:
                # 2.2 无法继续匹配
                res = False

        # 将当前结果记入备忘录
        self.memo[key] = res
        return res
```
