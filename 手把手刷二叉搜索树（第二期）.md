### 验证二叉搜索树
```
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isValidBST(self, root: TreeNode) -> bool:
        ## 限定以 root 为根的子树节点必须满足 max.val > root.val > min.val
        def is_valid_bst(root, min: TreeNode, max: TreeNode):
            ## base case
            if not root:
                return True
            ## 若 root.val 不符合 max 和 min 的限制，说明不是合法 BST
            if min and root.val <= min.val:
                return False
            if max and root.val >= max.val:
                return False
            ## 左子树的最大值为root  右子树的最小值为root
            return is_valid_bst(root.left, min, root) and is_valid_bst(root.right, root, max)
        return is_valid_bst(root,None,None)
```

### 二叉搜索树中的搜索
```
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def searchBST(self, root: TreeNode, val: int) -> TreeNode:
        if not root:
            return None
        if root.val == val:
            return root
        if root.val > val:
            return self.searchBST(root.left, val)
        if root.val < val:
            return self.searchBST(root.right, val)
```

### 二叉搜索树中的插入操作
```
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def insertIntoBST(self, root: TreeNode, val: int) -> TreeNode:
        if not root:
            return TreeNode(val)
        if root.val > val:
            root.left = self.insertIntoBST(root.left, val)
        if root.val < val:
            root.right = self.insertIntoBST(root.right, val)
        return root
```

### 删除二叉搜索树中的节点
```
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def deleteNode(self, root: TreeNode, key: int) -> TreeNode:
        if not root:
            return None
        if root.val == key:
            ## 处理 没有左右节点的情况
            if not root.left and not root.right:
                return None
            ## 处理 只有一个子节点的情况
            if not root.left and root.right:
                return root.right
            if root.left and not root.right:
                return root.left
            ## 处理 有两个子节点的情况
            ## 可以是左子树的最大值  也可以是右子树的最小值
            minNode = self.getMin(root.right)
            root.val = minNode.val
            root.right = self.deleteNode(root.right, minNode.val)
        elif root.val > key:
            root.left = self.deleteNode(root.left, key)
        else:
            root.right = self.deleteNode(root.right, key)
        return root

    def getMin(self, node):
        ## BST 最左边的就是最小的
        while node.left:
            node = node.left
        return node
```
