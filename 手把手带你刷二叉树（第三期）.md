### 二叉树的序列化与反序列化  前序遍历解法
```
# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Codec:

    def serialize(self, root):
        """Encodes a tree to a single string.
        
        :type root: TreeNode
        :rtype: str
        """
        def void_serialize(root, sb):
            if not root:
                sb.append("#,")
                return
            sb.append(str(root.val)+',')
            void_serialize(root.left, sb)
            void_serialize(root.right, sb)
        sb = []
        void_serialize(root, sb)
        return ''.join(sb)

        

    def deserialize(self, data):
        """Decodes your encoded data to tree.
        
        :type data: str
        :rtype: TreeNode
        """
        def func_deserialize(nodes):
            if len(nodes) == 0:
                return None
            rootVal = nodes.pop(0)
            if rootVal == "#":
                return None
            root = TreeNode(rootVal)
            root.left = func_deserialize(nodes)
            root.right = func_deserialize(nodes)
            return root
        nodes = data.split(',')
        return func_deserialize(nodes)
```

### 二叉树的序列化与反序列化  后序遍历解法
```
# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


## 中序遍历方法无法进行反序列化
## 要想实现反序列方法，首先要构造 root 节点。
## 前序遍历得到的 nodes 列表中，第一个元素是 root 节点的值；
## 后序遍历得到的 nodes 列表中，最后一个元素是 root 节点的值
## 中序遍历无法找到root节点的位置

## 后序遍历方法

class Codec:

    def serialize(self, root):
        """Encodes a tree to a single string.
        
        :type root: TreeNode
        :rtype: str
        """
        def void_serialize(root, sb):
            if not root:
                sb.append("#,")
                return
            void_serialize(root.left, sb)
            void_serialize(root.right, sb)
            sb.append(str(root.val)+",")
        sb = []
        void_serialize(root, sb)
        return ''.join(sb)

        

    def deserialize(self, data):
        """Decodes your encoded data to tree.
        
        :type data: str
        :rtype: TreeNode
        """
        def func_deserialize(nodes):
            if len(nodes) == 0:
                return None
            ## 从后往前取出元素
            rootVal = nodes.pop()
            if rootVal == "#":
                return None
            root = TreeNode(rootVal)
            ## 先构造右子树  后构造左子树
            root.right = func_deserialize(nodes)
            root.left = func_deserialize(nodes)
            return root
        ## '1,2,#,'这个字符串split之后会变成这样['1', '2', '#', '']
        nodes = data.split(',')[:-1]
        return func_deserialize(nodes)
        

# Your Codec object will be instantiated and called as such:
# ser = Codec()
# deser = Codec()
# ans = deser.deserialize(ser.serialize(root))
```

### 二叉树的序列化与反序列化  层级遍历解法
```
# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


## 层级遍历方法

class Codec:

    def serialize(self, root):
        """Encodes a tree to a single string.
        
        :type root: TreeNode
        :rtype: str
        """
        if not root:
            return ""
        sb = []
        q = []
        q.append(root)
        while len(q) > 0:
            cur = q.pop(0)
            if not cur:
                sb.append("#,")
                continue
            sb.append(str(cur.val)+',')
            q.append(cur.left)
            q.append(cur.right)
        return ''.join(sb)
        

        

    def deserialize(self, data):
        """Decodes your encoded data to tree.
        
        :type data: str
        :rtype: TreeNode
        """
        if not data:
            return None
        q = []
        nodes = data.split(',')
        ## 第一个元素就是root的值
        root = TreeNode(int(nodes[0]))
        ## 队列q记录父节点，将root加入队列
        q.append(root)
        i = 1
        while i < len(nodes)-2:
            ## 队列中存的都是父节点
            parent = q.pop(0)
            ## 父节点对应的左侧子节点的值
            left = nodes[i]
            if left != "#":
                parent.left = TreeNode(int(left))
                q.append(parent.left)
            else:
                parent.left = None

            ## 父节点对应的右侧子节点的值
            i += 1
            right = nodes[i]
            if right != "#":
                parent.right = TreeNode(int(right))
                q.append(parent.right)
            else:
                parent.right = None
            i += 1
        return root           
        

# Your Codec object will be instantiated and called as such:
# ser = Codec()
# deser = Codec()
# ans = deser.deserialize(ser.serialize(root))
```

### leetcode 652 寻找重复的子树
```
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right



class Solution:
    def findDuplicateSubtrees(self, root: TreeNode) -> List[TreeNode]:
        ##记录所有子树以及出现的次数
        memo = {}
        ##记录重复的子树根节点
        res = []
        def traverse(root):
            if not root:
                return "#"
            left = traverse(root.left)
            right = traverse(root.right)
            subTree = left + ',' + right + ',' + str(root.val)

            freq = memo.get(subTree, 0)
            ## 多次重复也只需要加入结果集一次 
            if freq == 1:
                res.append(root)
            ## 给子树对应的出现次数加一
            memo[subTree] = freq + 1
            return subTree
        traverse(root)
        return res
```
