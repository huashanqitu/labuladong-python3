### 回溯算法模板
```
result = []
def backtrack(路径, 选择列表):
    if 满足结束条件:
        result.add(路径)
        return
    for 选择 in 选择列表:
        做选择
        backtrack(路径, 选择列表)
        撤销选择
```

### 子集  数学归纳
```
class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:
        ## base case 返回一个空集
        if not nums:
            return [[]]
        ## 把最后一个元素拿出来
        n = nums.pop()
        ## 先递归算出前面元素的所有子集
        res = self.subsets(nums)

        size = len(res)
        for i in range(size):
            ## 在之前的结果追加
            tmp = res[i] + [n]
            res.append(tmp)
        return res
        
       
```

### 子集 回溯
```
class Solution:
    def __init__(self):
        self.res = []
    
    def subsets(self, nums: List[int]) -> List[List[int]]:
        ## 记录走过的路径
        track = []
        self.backtrack(nums, 0, track)
        return self.res

    def backtrack(self, nums, start, track):
        self.res.append(track+[])
        for i in range(start, len(nums)):
            ## 做选择
            track.append(nums[i])
            ## 回溯
            self.backtrack(nums, i+1, track)
            ## 撤销选择
            track.pop()
        

```

### 组合
```
class Solution:
    def __init__(self):
        self.res = []
    def combine(self, n: int, k: int) -> List[List[int]]:
        if k <= 0 or n <= 0:
            return self.res
        track = []
        self.backtrack(n, k, 1, track)
        return self.res

    def backtrack(self, n, k, start, track):
        ## 到达树的底部
        if k == len(track):
            self.res.append(track+[])
            return

        ## 注意i从start开始递增
        for i in range(start, n+1):
            ## 做选择
            track.append(i)
            self.backtrack(n, k, i+1, track)
            ## 撤销选择
            track.pop()    
```

### 排列
```
class Solution:
    def __init__(self):
        self.res = []

##  主函数 输入一组不重复的数字， 返回它们的全排列    
    def permute(self, nums: List[int]) -> List[List[int]]:
        ## 记录路径
        track = []
        self.backtrack(nums, track)
        return self.res

    def backtrack(self, nums, track):
        ## 触发结束条件
        if len(track) == len(nums):
            self.res.append(track+[])
            return

        for i in range(len(nums)):
            ## 排除不合法的选择
            if nums[i] in track:
                continue
            ## 做选择
            track.append(nums[i])
            ## 进入下一层决策树
            self.backtrack(nums, track)
            ## 取消选择
            track.pop()
        
```

```
class Solution:
    def __init__(self):
        self.res = []

##  主函数 输入一组不重复的数字， 返回它们的全排列    
    def permute(self, nums: List[int]) -> List[List[int]]:
        ## 记录路径
        track = []
        self.backtrack(nums, nums+[], track)
        return self.res

    def backtrack(self, nums, l, track):
        ## 触发结束条件
        if len(track) == len(nums):
            self.res.append(track+[])
            return

        for i in l:
            ## 做选择
            track.append(i)
            d = l + []
            d.remove(i)
            ## 进入下一层决策树
            self.backtrack(nums, d, track)
            ## 取消选择
            track.pop()
        
```
