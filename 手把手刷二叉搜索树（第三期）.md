### 不同的二叉搜索树
```
class Solution:
    def __init__(self):
        self.memo = []

    def numTrees(self, n: int) -> int:
        # 初始化备忘录的值
        self.memo = [[0]*(n+1) for _ in range(n+1)]

        def count(lo, hi):
            if lo > hi:
                return 1
            # 查备忘录
            if self.memo[lo][hi] != 0:
                return self.memo[lo][hi]
            res = 0
            for mid in range(lo, hi+1):
                left = count(lo, mid-1)
                right = count(mid+1, hi)
                res += left*right
            # 更新备忘录的值
            self.memo[lo][hi] = res
            return res
        
        return count(1, n)
```

### 不同的二叉搜索树 II  带备忘录
```
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def __init__(self):
        self.memo = {}

    def generateTrees(self, n: int) -> List[TreeNode]:
        return self.build(1, n)
    
    def build(self, lo, hi):
        res = []
        if lo > hi:
            res.append(None)
            return res
        if str(lo)+'_'+str(hi) in self.memo:
            return self.memo[str(lo)+'_'+str(hi)]
        for mid in range(lo, hi+1):
            left = self.build(lo, mid-1)
            right = self.build(mid+1, hi)
            for i in left:
                for j in right:
                    root = TreeNode(mid)
                    root.left = i
                    root.right = j
                    res.append(root)
        self.memo[str(lo)+'_'+str(hi)] = res
        return res
```
