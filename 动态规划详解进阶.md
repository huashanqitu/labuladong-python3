# 一  菲波那切数列
## 1 暴力递归
```
def fib(n):
  if n == 1 or n == 2:
    return 1
  else:
    return fib(n-1)+fib(n-2)
```

## 2 带备忘录的递归解法
```
def fib(n):
    if n<1:
        return 0
    memo = [0]*(n+1)
    return helper(memo,n)

def helper(memo,n):
    if 1<=n<=2:
        return 1
    if memo[n] != 0:
        return memo[n]
    memo[n] = helper(memo,n-1)+helper(memo,n-2)
    return memo[n]

```

## 3 dp数组的迭代解法
```
def fib(n):
    if n == 0:
        return 0
    if n == 1:
        return 1
    dp = [0] * (n+1)
    dp[1] = dp[2] = 1
    for i in range(3,n+1):
        dp[i] = dp[i-1]+dp[i-2]
    return dp[n]
```
##### 状态压缩
```
def fib(n):
    if n == 1 or n == 2:
        return 1
    prev,curr = 1,1
    for i in range(3,n+1):
        sum = prev + curr
        prev = curr
        curr = sum
    return curr
```

# 二 凑零钱问题

## 暴力解法
```
def coinChange(coins: list, amount: int):
    def dp(n):
        if n == 0:
            return 0
        if n < 0:
            return -1
        res = float('INF')
        for coin in coins:
            subproblem = dp(n-coin)
            if subproblem == -1:
                continue
            res = min(res, 1 + subproblem)

        return res if res != float('INF') else -1

    return dp(amount)
```

## 带备忘录的递归
```
def coinChange(coins: list, amount: int):
    memo = dict()
    def dp(n):
        if n in memo:
            return memo[n]
        if n == 0:
            return 0
        if n < 0:
            return -1
        res = float('INF')
        for coin in coins:
            subproblem = dp(n-coin)
            if subproblem == -1:
                continue
            res = min(res, 1+subproblem)
        memo[n] = res if res != float('INF') else -1
        return memo[n]
    return dp(amount)
```

## dp数组的迭代解法
```
def coinChange(coins: list, amount: int):
    dp = [amount+1]*(amount+1)
    dp[0] = 0
    for i in range(len(dp)):
        for coin in coins:
            if (i - coin) < 0:
                continue
            dp[i] = min(dp[i],1+dp[i-coin])
    if dp[amount] == (amount+1):
        return -1
    else:
        return dp[amount]
```
