### 寻找一个数（基本的二分搜索）
```
def binarySearch(nums, target):
    left = 0
    right = len(nums) - 1
    while left <= right:
        mid = left + (right - left) // 2
        if nums[mid] == target:
            return mid
        elif nums[mid] < target:
            left = mid + 1
        elif nums[mid] > target:
            right = mid - 1
    return -1
```

### 寻找左侧边界的二分搜索
```
def left_bound(nums, target):
    left, right = 0, len(nums) - 1
    while left <= right:
        mid = (left + right) // 2
        if nums[mid] == target:
            # 锁定左侧边界
            right = mid - 1
        elif nums[mid] < target:
            left = mid + 1
        elif nums[mid] > target:
            right = mid - 1
    # 检查left越界情况
    # 例子1  nums = [1]  target = 2    这种情况会返回1 超出了索引范围
    # 例子2  nums = [11, 12]  target = 3  这种情况会返回0 
    if left >= len(nums) or nums[left] != target:
        return -1
    return left
```

### 寻找右侧边界的二分搜索
```
def right_bound(nums, target):
    left, right = 0, len(nums) - 1
    while left <= right:
        mid = (left + right) // 2
        if nums[mid] == target:
            # 锁定右侧边界
            left = mid + 1
        elif nums[mid] < target:
            left = mid + 1
        elif nums[mid] > target:
            right = mid - 1
    # 检查right越界情况
    if right < 0 or nums[right] != target:
        return -1
    return right
```

### 875 爱吃香蕉的珂珂  寻找左侧边界
```
class Solution:
    def minEatingSpeed(self, piles: List[int], h: int) -> int:
        # 套用搜索左侧边界的算法框架
        left = 1
        right = self.getMax(piles) + 1
        while left < right:
            # 防止溢出
            mid = left + (right-left)//2
            if self.canFinish(piles, mid, h):
                right = mid
            else:
                left = mid + 1
        return left

    def canFinish(self, piles, speed, h):
        time = 0
        for n in piles:
            time += self.timeOf(n, speed)
        return time <= h

    def timeOf(self, n, speed):
        if n%speed > 0:
            return n//speed + 1
        else:
            return n//speed

    def getMax(self, piles):
        m = 0
        for n in piles:
            m = max(n, m)
        return m
```

### 分割数组的最大值
```
class Solution:
    def splitArray(self, nums: List[int], m: int) -> int:
        lo = max(nums)
        hi = sum(nums)
        for Max in range(lo,hi+1):
            # 如果最大子数组和是max
            # 至少可以把nums分割成n个子数组
            n = self.sp(nums, Max)
            # 为什么是<=不是==?
            if n <= m:
                return Max
        return -1

    # 辅助函数 若限制最大子数组和为Max
    # 计算nums至少可以被分割成几个子数组
    def sp(self, nums, Max):
        # 至少可以分割的子数组数量
        count = 1
        # 记录每个子数组的元素和
        sum = 0
        for i in range(0, len(nums)):
            if sum + nums[i] > Max:
                # 如果当前子数组和大于Max限制
                # 则这个子数组不能再添加元素了
                count += 1
                sum = nums[i]
            else:
                # 当前子数组和还没有达到Max限制
                # 还可以添加元素
                sum += nums[i]
        return count    
```

class Solution:
    def splitArray(self, nums: List[int], m: int) -> int:
        lo = max(nums)
        hi = sum(nums)
        while lo < hi:
            mid = lo + (hi - lo) // 2
            # 根据分割子数组的个数收缩搜索区间
            n = self.sp(nums, mid)
            if n == m:
                # 收缩右边界，达到搜索左边界的目的
                hi = mid
            elif n < m:
                # 最大子数组和上限高了，减小一些
                hi = mid
            elif n > m:
