### 最小覆盖子串
```
from collections import Counter
import collections

class Solution:
    def minWindow(self, s, t):
        start = 0
        minLen = float("inf")
        left = 0
        right = 0
        needs = Counter(t)
        window = collections.defaultdict(int)
        match = 0

        while right < len(s):
            c1 = s[right]
            if needs.__contains__(c1):
                window[c1] += 1
                if window[c1] == needs[c1]:
                    match += 1
            right += 1

            while match == len(needs):
                if right - left < minLen:
                    start = left
                    minLen = right - left
                c2 = s[left]
                if needs.__contains__(c2):
                    window[c2] -= 1
                    if window[c2] < needs[c2]:
                        match -= 1
                left += 1

        return "" if minLen == float("inf") else s[start:start+minLen]
```

### 567. 字符串的排列
```
class Solution:
    # 判断s中是否存在t的排列
    def checkInclusion(self, t: str, s: str) -> bool:
        need = {}
        window = {}
        for c in t:
            if c not in need:
                need[c] = 1
            else:
                need[c] += 1
        left = 0
        right = 0
        valid = 0
        while right < len(s):
            c = s[right]
            right += 1
            # 进行窗口内数据的一系列更新
            if c in need:
                if c not in window:
                    window[c] = 1
                else:
                    window[c] += 1
                if window[c] == need[c]:
                    valid += 1

            # 判断左侧窗口是否要收缩
            while (right - left) >= len(t):
                # 在这里判断是否找到了合法的子串
                if valid == len(need):
                    return True
                d = s[left]
                left += 1
                # 进行窗口内数据的一系列更新
                if d in need:
                    if window[d] == need[d]:
                        valid -= 1
                    window[d] -= 1
        
        # 未找到符合条件的子串
        return False
```

### 找到字符串中所有字母异位词

```
from collections import Counter
import collections

def findAnagrams(s, t):
    res = []
    left = 0
    right = 0
    needs = Counter(t)
    window = collections.defaultdict(int)
    match = 0

    while right < len(s):
        c1 = s[right]
        if needs.__contains__(c1):
            window[c1] += 1
            if window[c1] == needs[c1]:
                match += 1
        right += 1

        while match == len(needs):
            if right -left == len(t):
                res.append(left)
            c2 = s[left]
            if needs.__contains__(c2):
                windows[c2] -= 1
                if window[c2] < needs[c2]:
                    match -= 1
            left += 1

    return res
```

### 无重复字符的最长子串
```
from collections import Counter
import collections

def lengthOfLongestSubstring(s):
    left = 0
    right = 0
    window = collections.defaultdict(int)
    res = 0
    while right < len(s):
        c1 = s[right]
        window[c1] += 1
        right += 1
        while window[c1] > 1:
            c2 = s[left]
            window[c2] -= 1
            left += 1
        res = max(res, right - left)
    return res
```
