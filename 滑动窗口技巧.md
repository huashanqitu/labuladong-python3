### 最小覆盖子串
```
from collections import Counter
import collections

class Solution:
    def minWindow(self, s, t):
        start = 0
        minLen = float("inf")
        left = 0
        right = 0
        needs = Counter(t)
        window = collections.defaultdict(int)
        match = 0

        while right < len(s):
            c1 = s[right]
            if needs.__contains__(c1):
                window[c1] += 1
                if window[c1] == needs[c1]:
                    match += 1
            right += 1

            while match == len(needs):
                if right - left < minLen:
                    start = left
                    minLen = right - left
                c2 = s[left]
                if needs.__contains__(c2):
                    window[c2] -= 1
                    if window[c2] < needs[c2]:
                        match -= 1
                left += 1

        return "" if minLen == float("inf") else s[start:start+minLen]
```


### 找到字符串中所有字母异位词
```
from collections import Counter
import collections

def findAnagrams(s, t):
    res = []
    left = 0
    right = 0
    needs = Counter(t)
    window = collections.defaultdict(int)
    match = 0

    while right < len(s):
        c1 = s[right]
        if needs.__contains__(c1):
            window[c1] += 1
            if window[c1] == needs[c1]:
                match += 1
        right += 1

        while match == len(needs):
            if right -left == len(t):
                res.append(left)
            c2 = s[left]
            if needs.__contains__(c2):
                windows[c2] -= 1
                if window[c2] < needs[c2]:
                    match -= 1
            left += 1

    return res
```

### 无重复字符的最长子串
```
from collections import Counter
import collections

def lengthOfLongestSubstring(s):
    left = 0
    right = 0
    window = collections.defaultdict(int)
    res = 0
    while right < len(s):
        c1 = s[right]
        window[c1] += 1
        right += 1
        while window[c1] > 1:
            c2 = s[left]
            window[c2] -= 1
            left += 1
        res = max(res, right - left)
    return res
```
