### 动态规划解法
```
class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:
        dp = [1]*len(nums)
        for i in range(len(nums)):
            for j in range(i):
                if nums[i] > nums[j]:
                    dp[i] = max(dp[i],dp[j]+1)
        res = 0
        for i in range(len(dp)):
            res = max(res, dp[i])
        return res
```

### 二分查找解法 （就看看吧）
```
def lengthOfLIS(nums):
    top = [0] * len(nums)
    piles = 0
    for i in range(len(nums)):
        poker = nums[i]
        left = 0
        right = piles
        while left < right:
            mid = (left + right) // 2
            if top[mid] > poker:
                right = mid
            elif top[mid] < poker:
                left = mid + 1
            else:
                right = mid

        if left == piles:
            piles += 1
        top[left] = poker
    return piles
```
