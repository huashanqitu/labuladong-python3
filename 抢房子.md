```
class Solution:
    def rob(self, nums: List[int]) -> int:
        return self.dp(nums, 0)
##  返回 nums[start..] 能抢到的最大值
    def dp(self, nums, start):
        if start >= len(nums):
            return 0
##  比较         不抢 去下家           抢 去下下家
        res = max(self.dp(nums, start+1), nums[start]+self.dp(nums, start+2))
        return res
```

### 打家劫舍1 带备忘录
```
class Solution:

    def rob(self, nums: List[int]) -> int:
        memo = [-1]*len(nums)
        return self.dp(nums, 0, memo)

    def dp(self, nums, start, memo):
        if start >= len(nums):
            return 0
        if memo[start] != -1:
            return memo[start]
        res = max(self.dp(nums, start+1, memo), nums[start]+self.dp(nums, start+2, memo))
        memo[start] = res
        return res
```        

### 打家劫舍1 DP table
```
class Solution:
    def rob(self, nums: List[int]) -> int:
        n = len(nums)
##         dp[i] = x 表示：
##         从第 i 间房子开始抢劫，最多能抢到的钱为 x
##         base case: dp[n] = 0
        dp = [0]*(n+2)
        for i in range(n-1, -1, -1):
            dp[i] = max(dp[i+1],nums[i]+dp[i+2])
        return dp[0]
```        

### 打家劫舍2 环形
```
class Solution:
    def rob(self, nums: List[int]) -> int:
        n = len(nums)
        if n == 1:
            return nums[0]
        return max(self.robRange(nums,0,n-2), self.robRange(nums,1,n-1))

    def robRange(self, nums, start, end):
        n = len(nums)
        dp_i_1 = 0
        dp_i_2 = 0
        dp_i = 0
        for i in range(end, start-1, -1):
            dp_i = max(dp_i_1, nums[i]+dp_i_2)
            dp_i_2 = dp_i_1
            dp_i_1 = dp_i
        return dp_i
                       
```

### 打家劫舍3 
```
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
memo = {}
class Solution:

    def rob(self, root: TreeNode) -> int:
        if not root:
            return 0
        if memo.__contains__(root):
            return memo[root]
        if root.left:
            next_left = self.rob(root.left.left) + self.rob(root.left.right)
        else:
            next_left = 0
        if root.right:
            next_right = self.rob(root.right.left) + self.rob(root.right.right)
        else:
            next_right = 0
        do_it = root.val + next_left + next_right
        not_do = self.rob(root.left) + self.rob(root.right)
        res = max(do_it, not_do)
        memo[root] = res
        return res
```

### 打家劫舍3 dp状态代表抢和不抢可以获取的最大钱数
```
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def rob(self, root: TreeNode) -> int:
        res = self.dp(root)
        return max(res[0], res[1])

##    返回一个大小为 2 的数组 arr
##    arr[0] 表示不抢 root 的话，得到的最大钱数
##    arr[1] 表示抢 root 的话，得到的最大钱数

    def dp(self, root):
        if not root:
            return [0,0]
        left = self.dp(root.left)
        right = self.dp(root.right)
        do_it = root.val + left[0] + right[0]
        not_do = max(left[0],left[1]) + max(right[0], right[1])
        return [not_do, do_it]
```
