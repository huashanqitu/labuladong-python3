### 滑动窗口最大值

```
class Solution:
    def maxSlidingWindow(self, nums: List[int], k: int) -> List[int]:
        window = MonotonicQueue()
        res = []
        for i in range(len(nums)):
            if i < k - 1:
                # 先把窗口的前k-1填满
                window.push(nums[i])
            else:
                # 窗口开始向前滑动
                # 移入新元素
                window.push(nums[i])
                # 将当前窗口中的最大元素记入结果
                res.append(window.max())
                # 移除最后的元素
                window.pop(nums[i-k+1])
        # 返回res
        return res

# 单调队列
class MonotonicQueue:
    def __init__(self):
        self.q = []
    def push(self, n):
        # 将小于n的元素全部删除
        while len(self.q) > 0 and self.q[-1] < n:
            self.q.pop()
        # 然后将n加入尾部
        self.q.append(n)

    def max(self):
        return self.q[0]

    def pop(self, n):
        if n == self.q[0]:
            self.q.pop(0)
```
