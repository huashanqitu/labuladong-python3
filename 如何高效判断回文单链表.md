### 验证回文串
```
class Solution:
    def isPalindrome(self, s: str) -> bool:
        s = ''.join([x for x in s if x.isalnum()])
        s = s.lower()

        n = len(s)
        if n == 1:
            return True
        left = 0
        right = n - 1
        while left < right:
            if s[left] == s[right]:
                left += 1
                right -= 1
            else:
                return False
        return True
```

### 判断回文单链表
```
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def __init__(self):
        self.left = None
        
    def isPalindrome(self, head) -> bool:
        self.left = head
        return self.traverse(head)

    def traverse(self, right):
        if not right:
            return True
        ## 后续遍历代码
        res = self.traverse(right.next)
        if res==True and right.val == self.left.val:
            self.left = self.left.next
        else:
            return False
        return res
```
